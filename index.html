Game Over."; document.getElementById("leftBtn").style.display = "none"; document.getElementById("rightBtn").style.display = "none"; } function goLeft() { document.getElementById("story").innerText = "You found a path deeper into the forest. Do you go Forward or Back?"; document.getElementById("leftBtn").innerText = "Go Forward"; document.getElementById("leftBtn").setAttribute("onclick", "goForward()"); document.getElementById("rightBtn").innerText = "Go Back"; document.getElementById("rightBtn").setAttribute("onclick", "goBack()"); } function goRight() { document.getElementById("story").innerText = "A monster appears! Do you Fight or Run?"; document.getElementById("leftBtn").innerText = "Fight"; document.getElementById("leftBtn").setAttribute("onclick", "fightMonster()"); document.getElementById("rightBtn").innerText = "Run"; document.getElementById("rightBtn").setAttribute("onclick", "runAway()"); }
